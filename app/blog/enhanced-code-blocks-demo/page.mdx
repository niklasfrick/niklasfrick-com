  children: React.ReactNode
  variant?: 'primary' | 'secondary'
  onClick?: () => void
}

  return (
    <button

## Python Script Example

```python
# data_processor.py
import pandas as pd
import numpy as np
from typing import List, Dict

def process_data(data: List[Dict]) -> pd.DataFrame:
    """
    Process raw data into a pandas DataFrame
    
    Args:
        data: List of dictionaries containing raw data
        
    Returns:
        Processed DataFrame
    """
    df = pd.DataFrame(data)
    
    # Clean missing values
    df = df.dropna()
    
    # Convert date columns
    date_columns = ['created_at', 'updated_at']
    for col in date_columns:
        if col in df.columns:
            df[col] = pd.to_datetime(df[col])
    
    return df

def calculate_statistics(df: pd.DataFrame) -> Dict:
    """Calculate basic statistics for numeric columns"""
    numeric_cols = df.select_dtypes(include=[np.number]).columns
    
    stats = {}
    for col in numeric_cols:
        stats[col] = {
            'mean': df[col].mean(),
            'std': df[col].std(),
            'min': df[col].min(),
            'max': df[col].max()
        }
    
    return stats
```

    COUNT(*) as total_users,

## Features

The enhanced code blocks include:

- **File names**: Display the filename in the header
- **Syntax highlighting**: Proper syntax highlighting for various languages
- **Line numbers**: Optional line numbers for better code readability
- **Copy button**: One-click copy functionality with visual feedback
- **File type indicators**: Shows the file extension in the header
- **Dark mode support**: Automatically adapts to your theme
- **Responsive design**: Works well on all screen sizes

## Usage

To use enhanced code blocks in your MDX files, simply add a comment with the filename at the beginning of your code block:

```markdown
```javascript
// my-file.js
const example = "This will show 'my-file.js' in the header";
```
```

The filename comment will be automatically detected and displayed in the header, while being removed from the actual code content.
